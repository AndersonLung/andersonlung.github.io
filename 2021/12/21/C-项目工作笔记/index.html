    管理所有的错误码

    设错误码和含义的JSON 对应

    `
    const char *strerror(int err){
    switch (err)
    {
    #define XX(code, text)
    case ##code:
    return text;
    COMMON_CODE_MAP(XX)
    OLD_CODE_MAP(XX);
    ERROR_CODE_MAP(XX)
    SLAM_CODE_MAP(XX)
    ELEVATOR_CODE_MAP(XX)
    #undef XX
    }
    return “unknown error”;
    }
    }

    #define SET_CODE(code)
    res[“code”] = code;
    res[“msg”] = strerror(code);
    原理：构建错误码不同的map，然后根据map来找到对应的错误码原因来实现代码只存在错误码，不会直接引用原因 错误码的map 构建：
    #define COMMON_CODE_MAP(XX) XX(SUCCESS, “success”)
    对错误码开始的定义：
    enum
    {
    #define XX(code, _) _##code,
    COMMON_CODE_MAP(XX)

    CODE_BEGIN = 11010000 - 1,
    UOLD_CODE_MAP(XX) UERROR_CODE_MAP(XX)
    #undef XX
    };
    `

    对步骤拆解善用| & 运算符

    如：
    可以定义步骤为
    enum { BEHAVIOR_TYPE_NONE = 0, BEHAVIOR_TYPE_REPORT = (1 << 0), BEHAVIOR_TYPE_RECORD = (1 << 1), BEHAVIOR_TYPE_LIST = (1 << 2), BEHAVIOR_TYPE_DOCK = (1 << 3), BEHAVIOR_TYPE_STAT= (1 << 4), BEHAVIOR_TYPE_NAV = (1 << 5), BEHAVIOR_TYPE_DIS_IMG = (1 << 6), BEHAVIOR_TYPE_AUTO_DOCK = (1 << 7), };

    代码

    `
    bool CUvInsideApiOld::SyncMsgSendAndWait(const CMsg& send, CSyncMsg& msg, const CMsg& cancel, UV_STATUS_TYPE status)
    {
    if (CheckAndGetCancelStatus(send, msg, status))
    {
    return true;
    }
    bool rt = SyncMsgSendAndWait(send.type, send.content, send.timeout, msg);
    CheckAndGetCancelStatus(send, msg, status);
    if (cancel.type != UV_MSG_NONE && msg.NeedCancel())
    { //处理中断消息
    CSyncMsg cancel_msg;
    UV_MSG_TYPE msgtype = cancel.type;
    if (msg.result == UV_PAUSE && cancel.type == UV_MSG_STOP_AUTO_MAPPING)
    { //兼容处理 探索暂停，发送暂停指令
    Json::Value content;
    content[“cmd”] = “suspend”;
    msgtype = UV_MSG_CONTROL_AUTO_MAPPING;
    SyncMsgSendAndWait(UV_MSG_CONTROL_AUTO_MAPPING, content, cancel.timeout, cancel_msg);
    }
    else
    {
    SyncMsgSendAndWait(cancel.type, cancel.content, cancel.timeout, cancel_msg);
    }
    ROS_INFO_STREAM(“[CUvInsideApiOld::SyncMsgSendAndWait] cancel title=”

    }
    return rt;
    }

    这里的流程是，在请求某个子任务时候，先第一次请求取消状态，然后第一次请求，这里是阻塞的，获得结果，需要再次检查取消状态，如果是需要取消的，发送取消命令，让其他模块终止刚刚工作 bool CUvInsideApiOld::SyncMsgSendAndWait(UV_MSG_TYPE type, const Json::Value& content, long msec, CSyncMsg& msg) { ros::Publisher* pub = nullptr; if (type != UV_MSG_NOTHING && type != UV_MSG_PAUSE && type != UV_MSG_SAFE_STRATEGY) { pub = GetPublisher(type); if (pub == nullptr) { ROS_INFO_STREAM("[CUvInsideApiOld::SyncMsgSendAndWait] type=" << type << " invalid"); msg.result = UV_FAILED; return false; } } msg.id = GenID() | GID_GET_TYPE_BIT(type) | GID_GET_FLAG_BIT(); std_msgs::String data; if (pub != nullptr) { Json::FastWriter writer; Json::Value root; writer.omitEndingLineFeed(); if (!content.isNull()) { root["content"] = content; } root["content"]["id"] = msg.id; root["title"] = Enum2String(type); data.data = writer.write(root); ROS_INFO_STREAM("[CUvInsideApiOld::SyncMsgSendAndWait] pub msg=" << data.data); } { boost::mutex::scoped_lock lock(mutex_); if (msg.flags != 0 && safe_flags.MatchLock(msg.flags)) { msg.result = UV_SAFEGUARD; msg.message = "safeguard" + safe_flags.GetLockName(msg.flags); ROS_INFO_STREAM("[CUvInsideApiOld::SyncMsgSendAndWait] flags=" << msg.flags << ",matched=" << safe_flags.MatchLock(msg.flags) << " safeguard"); return false; } safe_flags.SetWaitFlags(msg.flags); sync_q_.push_back(msg); if (pub) { pub->publish(data); } if (msec == -1) { cond_.wait(lock); } else if (cond_.timed_wait(lock, boost::posix_time::millisec(msec)) == false) { msg.result = UV_TIMEOUT; msg.message = Enum2String(type) + std::string(":timeout"); SyncMsgDel(type); safe_flags.ClearWaitFlags(); return false; } safe_flags.ClearWaitFlags(); return SyncMsgGet(type, msg); } }
    这个函数是上一个函数的子流程，专门用来处理发送数据，主要精妙之处在于利用 boost::condition_variable 的wait 函数和timed_wait 来一直占有锁，如果收到取消，暂停，回复，超时等都会
    `

    bool CUvInsideApiOld::AsyncMsgSendAndSpinWait(UV_MSG_TYPE type, const Json::Value& content, long msec, Json::Value& res)
    {
    ros::Publisher* pub = GetPublisher(type);
    if (pub == nullptr)
    {
    ROS_INFO_STREAM(“[CUvInsideApiOld::AsyncMsgSendAndSpinWait] type=” << type << “ invalid”);
    UV_SET_CODE(UV_IMPEDED);
    return false;
    }

    int64_t gid = GenID() | GID_GET_TYPE_BIT(type) | GID_GET_SPIN_BIT();
    Json::Value root;
    root[“title”] = Enum2String(type);
    root[“content”] = content;
    root[“content”][“id”] = gid;

    if (spin_q_.size() > 8)
    {
    UV_SET_CODE(UV_ESYSBUSY);
    return false;
    }

    spin_q_.push_back(CSpinMsg(gid, &res));
    {
    Json::FastWriter write;
    write.omitEndingLineFeed();
    std_msgs::String data;
    data.data = write.write(root);
    pub->publish(data);
    }

    ros::Rate rate(1000);
    while (ros::ok())
    {
    –msec;
    ros::spinOnce();
    auto iter = std::find(spin_q_.begin(), spin_q_.end(), gid);
    if (iter == spin_q_.end() || msec <= 1)
    {
    UV_SET_CODE(UV_TIMEOUT);
    if (iter != spin_q_.end())
    spin_q_.erase(iter);
    break;
    }
    if (iter->get_respond())
    {
    spin_q_.erase(iter);
    break;
    }
    rate.sleep();
    }
    if (spin_q_.size() > 0)
    ROS_INFO_STREAM(“[CUvInsideApiOld::AsyncMsgSendAndSpinWait] size=” << spin_q_.size());
    return true;
    }

    任务的暂停、恢复、取消

    取消流程: 任务开启后，先判断是否有取消标志，则直接取消，没有继续运行，如果中途取消。condition_variable.notify_one() 则可以解除lock ，再次调用检查，确定是否是其他异常流程，启动异常流程

    暂停恢复流程：
    如取消设置暂停标志，等待后续安全事件解除后恢复任务，也可以将该任务的状态职位pause ，待下次进行任务检测，计算超时，超时时间内就可以重启让他启动
